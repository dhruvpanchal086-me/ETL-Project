CREATE PROCEDURE dwh.usp_load_dim_advisor
AS
BEGIN
    SET NOCOUNT ON;

    -- UPDATE existing records if anything changed (SCD Type 1 - overwrite)
    UPDATE dwh.dim_advisor
    SET
        advisor_name        = src.AdvisorName,
        region              = src.Region,
        join_date           = CAST(src.JoinDate AS DATE),
        experience_years    = CAST(src.ExperienceYears AS INT),
        dwh_updated_date    = GETDATE()
    FROM dwh.dim_advisor tgt
    INNER JOIN stg.stg_advisors src
        ON tgt.advisor_id = src.AdvisorID
    WHERE
        tgt.advisor_name        != src.AdvisorName OR
        tgt.region              != src.Region OR
        tgt.experience_years    != CAST(src.ExperienceYears AS INT);

    -- INSERT new records that don't exist yet
    INSERT INTO dwh.dim_advisor (
        advisor_id,
        advisor_name,
        region,
        join_date,
        experience_years,
        dwh_created_date,
        dwh_updated_date,
        is_active
    )
    SELECT
        src.AdvisorID,
        src.AdvisorName,
        src.Region,
        CAST(src.JoinDate AS DATE),
        CAST(src.ExperienceYears AS INT),
        GETDATE(),
        GETDATE(),
        'Y'
    FROM stg.stg_advisors src
    WHERE NOT EXISTS (
        SELECT 1 FROM dwh.dim_advisor tgt
        WHERE tgt.advisor_id = src.AdvisorID
    );

END;

CREATE PROCEDURE dwh.usp_load_dim_client
AS
BEGIN
    SET NOCOUNT ON;

    -- EXPIRE old records where data has changed (SCD Type 2)
    UPDATE dwh.dim_client
    SET
        effective_end_date  = CAST(GETDATE() AS DATE),
        is_current          = 'N',
        dwh_updated_date    = GETDATE()
    FROM dwh.dim_client tgt
    INNER JOIN stg.stg_clients src
        ON tgt.client_id = src.ClientID
        AND tgt.is_current = 'Y'
    WHERE
        tgt.segment             != src.Segment OR
        tgt.risk_profile        != ISNULL(NULLIF(src.RiskProfile, ''), 'Unknown') OR
        tgt.country             != src.Country OR
        tgt.primary_advisor_id  != src.PrimaryAdvisorID;

    -- INSERT new version for changed records
    INSERT INTO dwh.dim_client (
        client_id,
        client_name,
        segment,
        risk_profile,
        primary_advisor_id,
        client_since,
        country,
        effective_start_date,
        effective_end_date,
        is_current,
        dwh_created_date,
        dwh_updated_date
    )
    SELECT
        src.ClientID,
        src.ClientName,
        src.Segment,
        ISNULL(NULLIF(src.RiskProfile, ''), 'Unknown'),
        src.PrimaryAdvisorID,
        CAST(src.ClientSince AS DATE),
        src.Country,
        CAST(GETDATE() AS DATE),
        NULL,
        'Y',
        GETDATE(),
        GETDATE()
    FROM stg.stg_clients src
    INNER JOIN dwh.dim_client tgt
        ON tgt.client_id = src.ClientID
        AND tgt.is_current = 'N'
        AND tgt.effective_end_date = CAST(GETDATE() AS DATE)
    WHERE NOT EXISTS (
        SELECT 1 FROM dwh.dim_client cur
        WHERE cur.client_id = src.ClientID
        AND cur.is_current = 'Y'
    );

    -- INSERT brand new clients
    INSERT INTO dwh.dim_client (
        client_id,
        client_name,
        segment,
        risk_profile,
        primary_advisor_id,
        client_since,
        country,
        effective_start_date,
        effective_end_date,
        is_current,
        dwh_created_date,
        dwh_updated_date
    )
    SELECT
        src.ClientID,
        src.ClientName,
        src.Segment,
        ISNULL(NULLIF(src.RiskProfile, ''), 'Unknown'),
        src.PrimaryAdvisorID,
        CAST(src.ClientSince AS DATE),
        src.Country,
        CAST(GETDATE() AS DATE),
        NULL,
        'Y',
        GETDATE(),
        GETDATE()
    FROM stg.stg_clients src
    WHERE NOT EXISTS (
        SELECT 1 FROM dwh.dim_client tgt
        WHERE tgt.client_id = src.ClientID
    );

END;

CREATE PROCEDURE dwh.usp_load_fact_holding
AS
BEGIN
    SET NOCOUNT ON;

    -- TRUNCATE fact table before reload (full refresh)
    TRUNCATE TABLE dwh.fact_holding;

    INSERT INTO dwh.fact_holding (
        account_sk,
        security_sk,
        date_sk,
        account_id,
        security_id,
        valuation_date,
        quantity,
        market_value
    )
    SELECT
        da.account_sk,
        ds.security_sk,
        dd.date_sk,
        src.AccountID,
        src.SecurityID,
        CAST(src.ValuationDate AS DATE),
        CAST(src.Quantity AS DECIMAL(18,4)),
        CAST(src.MarketValue AS DECIMAL(18,2))
    FROM stg.stg_holdings src
    INNER JOIN dwh.dim_account da
        ON da.account_id = src.AccountID
        AND da.is_current = 'Y'
    INNER JOIN dwh.dim_security ds
        ON ds.security_id = src.SecurityID
        AND ds.is_active = 'Y'
    INNER JOIN dwh.dim_date dd
        ON dd.full_date = CAST(src.ValuationDate AS DATE);

END;

CREATE PROCEDURE dwh.usp_load_fact_trade
AS
BEGIN
    SET NOCOUNT ON;

    TRUNCATE TABLE dwh.fact_trade;

    INSERT INTO dwh.fact_trade (
        account_sk,
        security_sk,
        trade_date_sk,
        settle_date_sk,
        trade_id,
        account_id,
        security_id,
        trade_date,
        settle_date,
        side,
        price,
        quantity,
        amount,
        settle_lag_days
    )
    SELECT
        da.account_sk,
        ds.security_sk,
        dd_trade.date_sk,
        dd_settle.date_sk,
        src.TradeID,
        src.AccountID,
        src.SecurityID,
        CAST(src.TradeDate AS DATE),
        CAST(src.SettleDate AS DATE),
        src.Side,
        CAST(src.Price AS DECIMAL(18,4)),
        CAST(src.Quantity AS DECIMAL(18,4)),
        CAST(src.Amount AS DECIMAL(18,2)),
        DATEDIFF(DAY,
            CAST(src.TradeDate AS DATE),
            CAST(src.SettleDate AS DATE))
    FROM stg.stg_trades src
    INNER JOIN dwh.dim_account da
        ON da.account_id = src.AccountID
        AND da.is_current = 'Y'
    INNER JOIN dwh.dim_security ds
        ON ds.security_id = src.SecurityID
        AND ds.is_active = 'Y'
    INNER JOIN dwh.dim_date dd_trade
        ON dd_trade.full_date = CAST(src.TradeDate AS DATE)
    INNER JOIN dwh.dim_date dd_settle
        ON dd_settle.full_date = CAST(src.SettleDate AS DATE);

END;

CREATE PROCEDURE dwh.usp_load_fact_performance
AS
BEGIN
    SET NOCOUNT ON;

    TRUNCATE TABLE dwh.fact_performance;

    INSERT INTO dwh.fact_performance (
        account_sk,
        benchmark_sk,
        date_sk,
        account_id,
        benchmark_code,
        month_start,
        return_pct,
        benchmark_return_pct,
        excess_return_pct,
        cumulative_return_pct
    )
    SELECT
        da.account_sk,
        db.benchmark_sk,
        dd.date_sk,
        src.AccountID,
        src.BenchmarkCode,
        CAST(src.MonthStart AS DATE),
        CASE WHEN src.ReturnPct = '' OR src.ReturnPct IS NULL
             THEN NULL
             ELSE CAST(src.ReturnPct AS DECIMAL(10,6)) END,
        CAST(src.BenchmarkReturnPct AS DECIMAL(10,6)),
        CAST(src.ExcessReturnPct AS DECIMAL(10,6)),
        CAST(src.CumulativeReturnPct AS DECIMAL(10,6))
    FROM stg.stg_performance src
    INNER JOIN dwh.dim_account da
        ON da.account_id = src.AccountID
        AND da.is_current = 'Y'
    INNER JOIN dwh.dim_benchmark db
        ON db.benchmark_code = src.BenchmarkCode
        AND db.is_active = 'Y'
    INNER JOIN dwh.dim_date dd
        ON dd.full_date = CAST(src.MonthStart AS DATE);

END;

-- Cashflow sign check
SELECT
    cashflow_type,
    COUNT(*)                                        AS total_rows,
    SUM(CASE WHEN amount > 0 THEN 1 ELSE 0 END)    AS positive_count,
    SUM(CASE WHEN amount < 0 THEN 1 ELSE 0 END)    AS negative_count
FROM dwh.fact_cashflow
GROUP BY cashflow_type
ORDER BY cashflow_type;


-- Trade side check
SELECT
    side,
    COUNT(*)                                        AS total_rows,
    SUM(CASE WHEN amount > 0 THEN 1 ELSE 0 END)    AS positive_amount,
    SUM(CASE WHEN amount < 0 THEN 1 ELSE 0 END)    AS negative_amount
FROM dwh.fact_trade
GROUP BY side;

    COUNT(*)                                                AS total_rows,
    SUM(CASE WHEN close_price IS NULL THEN 1 ELSE 0 END)   AS null_prices,
    SUM(CASE WHEN close_price IS NOT NULL THEN 1 ELSE 0 END) AS valid_prices
FROM dwh.fact_security_price;



-- NULL return pct
SELECT
    COUNT(*)                                                AS total_rows,
    SUM(CASE WHEN return_pct IS NULL THEN 1 ELSE 0 END)    AS null_returns,
    SUM(CASE WHEN return_pct IS NOT NULL THEN 1 ELSE 0 END) AS valid_returns
FROM dwh.fact_performance;


-- Negative settle lag
SELECT COUNT(*) AS negative_settle_lag_count
FROM dwh.fact_trade
WHERE settle_lag_days < 0;


-- Total AUM by currency
SELECT
    da.base_currency,
    COUNT(DISTINCT fh.account_id)       AS total_accounts,
    SUM(fh.market_value)                AS total_aum,
    AVG(fh.market_value)                AS avg_holding_value
FROM dwh.fact_holding fh
INNER JOIN dwh.dim_account da
    ON da.account_sk = fh.account_sk
GROUP BY da.base_currency
ORDER BY total_aum DESC;

ALTER PROCEDURE dwh.usp_load_fact_trade

AS
BEGIN
    SET NOCOUNT ON;

    TRUNCATE TABLE dwh.fact_trade;

    INSERT INTO dwh.fact_trade (
        account_sk,
        security_sk,
        trade_date_sk,
        settle_date_sk,
        trade_id,
        account_id,
        security_id,
        trade_date,
        settle_date,
        side,
        price,
        quantity,
        amount,
        settle_lag_days
    )
    SELECT
        da.account_sk,
        ds.security_sk,
        dd_trade.date_sk,
        dd_settle.date_sk,
        src.TradeID,
        src.AccountID,
        src.SecurityID,
        CAST(src.TradeDate  AS DATE),
        CAST(src.SettleDate AS DATE),
        src.Side,
        CAST(src.Price      AS DECIMAL(18,4)),
        CAST(src.Quantity   AS DECIMAL(18,4)),
        CAST(src.Amount     AS DECIMAL(18,2)),
        DATEDIFF(DAY,
            CAST(src.TradeDate  AS DATE),
            CAST(src.SettleDate AS DATE))
    FROM (
        -- STEP 1: Remove exact duplicate rows first
        SELECT DISTINCT
            TradeID, AccountID, SecurityID,
            TradeDate, SettleDate, Side,
            Price, Quantity, Amount
        FROM stg.stg_trades
    ) all_trades
    -- STEP 2: For same TradeID with different amounts,
    --         keep the row where amount matches price x quantity
    INNER JOIN (
        SELECT
            TradeID,
            MIN(ABS(CAST(Amount AS DECIMAL(18,2)) -
                ROUND(CAST(Price AS DECIMAL(18,4)) *
                      CAST(Quantity AS DECIMAL(18,4)), 2)
            )) AS min_diff
        FROM (
            SELECT DISTINCT TradeID, Price, Quantity, Amount
            FROM stg.stg_trades
        ) d
        GROUP BY TradeID
    ) best ON all_trades.TradeID = best.TradeID
           AND ABS(CAST(all_trades.Amount AS DECIMAL(18,2)) -
               ROUND(CAST(all_trades.Price AS DECIMAL(18,4)) *
                     CAST(all_trades.Quantity AS DECIMAL(18,4)), 2)
               ) = best.min_diff
    -- Rename for clarity
    CROSS APPLY (SELECT all_trades.TradeID, all_trades.AccountID,
        all_trades.SecurityID, all_trades.TradeDate, all_trades.SettleDate,
        all_trades.Side, all_trades.Price, all_trades.Quantity,
        all_trades.Amount) src
    INNER JOIN dwh.dim_account da
        ON da.account_id = src.AccountID
        AND da.is_current = 'Y'
    INNER JOIN dwh.dim_security ds
        ON ds.security_id = src.SecurityID
        AND ds.is_active  = 'Y'
    INNER JOIN dwh.dim_date dd_trade
        ON dd_trade.full_date = CAST(src.TradeDate AS DATE)
    INNER JOIN dwh.dim_date dd_settle
        ON dd_settle.full_date = CAST(src.SettleDate AS DATE);

END;

BEGIN

    INSERT INTO dwh.fact_trade (
        account_sk,
        security_sk,
        trade_date_sk,
        settle_date_sk,
        trade_id,
        account_id,
        security_id,
        trade_date,
        settle_date,
        side,
        price,
        quantity,
        amount,
        settle_lag_days
    )
    SELECT
        da.account_sk,
        ds.security_sk,
        dd_trade.date_sk,
        dd_settle.date_sk,
        src.TradeID,
        src.AccountID,
        src.SecurityID,
        CAST(src.TradeDate  AS DATE),
        CAST(src.SettleDate AS DATE),
        src.Side,
        CAST(src.Price      AS DECIMAL(18,4)),
        CAST(src.Quantity   AS DECIMAL(18,4)),
        CAST(src.Amount     AS DECIMAL(18,2)),
        DATEDIFF(DAY,
            CAST(src.TradeDate  AS DATE),
            CAST(src.SettleDate AS DATE))
    FROM (
        -- FINAL: One row per TradeID
        -- For amended trades: keep row closest to price x quantity
        -- For exact duplicates: keep just one row
        SELECT
            TradeID, AccountID, SecurityID,
            TradeDate, SettleDate, Side,
            Price, Quantity, Amount
        FROM (
            SELECT
                TradeID, AccountID, SecurityID,
                TradeDate, SettleDate, Side,
                Price, Quantity, Amount,
                ROW_NUMBER() OVER (
                    PARTITION BY TradeID
                    ORDER BY
                        ABS(CAST(Amount AS DECIMAL(18,2)) -
                            ROUND(CAST(Price AS DECIMAL(18,4)) *
                                  CAST(Quantity AS DECIMAL(18,4)), 2)
                        ) ASC
                ) AS rn
            FROM (
                -- Remove fully identical rows first
                SELECT DISTINCT
                    TradeID, AccountID, SecurityID,
                    TradeDate, SettleDate, Side,
                    Price, Quantity, Amount
                FROM stg.stg_trades
            ) deduped
        ) ranked
        WHERE rn = 1
    ) src
    INNER JOIN dwh.dim_account da
        ON da.account_id = src.AccountID
        AND da.is_current = 'Y'
    INNER JOIN dwh.dim_security ds
        ON ds.security_id = src.SecurityID
        AND ds.is_active  = 'Y'
    INNER JOIN dwh.dim_date dd_trade
        ON dd_trade.full_date = CAST(src.TradeDate AS DATE)
    INNER JOIN dwh.dim_date dd_settle
        ON dd_settle.full_date = CAST(src.SettleDate AS DATE);

END;
